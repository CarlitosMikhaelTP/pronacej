	id SERIAL PRIMARY KEY,
	type_user_id INT NOT NULL,

	last_name VARCHAR(50) NOT NULL,
	email VARCHAR(50) NOT NULL,
	password VARCHAR(50) NOT NULL,

	FOREIGN KEY (created_by) REFERENCES person(id),
